### 2.2 审批流程引擎

```python
class ApprovalFlowEngine:
    """审批流程引擎"""
    
    def __init__(self, chain_repo: ChainRepository, request_repo: RequestRepository):
        self.chain_repo = chain_repo
        self.request_repo = request_repo
    
    def process_submit(self, business_type: str, applicant_id: int, content: dict) -> ApprovalRequest:
        """处理申请提交"""
        # 1. 获取对应业务类型的审批链
        chain = self.chain_repo.get_by_type(business_type)
        
        # 2. 创建审批申请记录
        request = self.request_repo.create(...)
        
        # 3. 触发第一个审批节点
        self._trigger_next_node(request, chain.nodes[0])
        
        return request
    
    def process_approval(self, request_id: int, approver_id: int, action: str, comment: str):
        """处理审批操作"""
        request = self.request_repo.get_by_id(request_id)
        current_node = request.current_node
        
        # 1. 记录审批操作
        self._record_action(request, approver_id, action, comment)
        
        if action == 'reject':
            # 拒绝：直接结束流程
            self._reject_request(request)
        else:
            # 通过：判断审批模式
            if current_node.approval_mode == 'or':
                # 或签：直接进入下一节点
                self._move_to_next_node(request)
            elif current_node.approval_mode == 'and':
                # 会签：检查是否所有人都已审批
                if self._is_all_approved(request, current_node):
                    self._move_to_next_node(request)
                else:
                    # 等待其他人审批
                    self._wait_for_others(request)
    
    def _move_to_next_node(self, request: ApprovalRequest):
        """移动到下一审批节点"""
        chain = request.chain
        current_order = request.current_node.node_order
        
        next_node = chain.nodes.filter(node_order=current_order + 1).first()
        
        if next_node:
            # 有下一节点
            request.current_node = next_node
            request.current_approver = self._resolve_approver(next_node)
            request.status = 'pending'
        else:
            # 无下一节点，审批完成
            request.status = 'approved'
            request.completed_at = now()
    
    def _resolve_approver(self, node: ApprovalNode) -> User:
        """解析审批人"""
        if node.node_type == 'role':
            # 按角色：查询该角色的第一个用户
            return self._get_user_by_role(node.node_value)
        elif node.node_type == 'user':
            # 指定用户
            return User.objects.get(id=node.node_value)
        elif node.node_type == 'department_head':
            # 部门负责人
            return self._get_department_head(request.applicant)
```

class MessageService:
    """消息服务"""
    
    def __init__(self, message_repo: MessageRepository, notification_client: NotificationClient):
        self.message_repo = message_repo
        self.notification_client = notification_client
    
    def send_approval_notification(self, request: ApprovalRequest, event: str):
        """发送审批通知"""
        # 构建消息内容
        message = self._build_message(request, event)
        
        # 保存消息到数据库
        self.message_repo.create(message)
        
        # 发送实时通知
        self.notification_client.push(message)
    
    def send_feedback_notification(self, feedback: Feedback, event: str):
        """发送反馈通知"""
        message = self._build_feedback_message(feedback, event)
        self.message_repo.create(message)
        self.notification_client.push(message)
    
    def get_unread_count(self, user_id: int) -> int:
        """获取未读消息数量"""
        return self.message_repo.count_unread(user_id)
    
    def mark_as_read(self, message_id: int, user_id: int) -> None:
        """标记消息已读"""
        self.message_repo.mark_read(message_id, user_id)
    
    def get_messages(self, user_id: int, filters: MessageFilters) -> List[Message]:
        """获取消息列表"""
        return self.message_repo.get_by_user(user_id, filters)

class FeedbackService:

        # 通知管理员
        self._notify_admin(feedback)
        
        return feedback
    
    def get_list(self, user_id: int, filters: FeedbackFilters) -> List[Feedback]:
        """获取反馈列表"""
        return Feedback.objects.filter(user_id=user_id).filter(...)
    
    def reply(self, feedback_id: int, handler_id: int, content: str) -> Feedback:
        """回复反馈（管理员）"""
        feedback = Feedback.objects.get(id=feedback_id)
        feedback.handler_id = handler_id
        feedback.reply_content = content
        feedback.replied_at = now()
        feedback.status = 'replied'
        feedback.save()
        
        # 通知用户
        self._notify_user(feedback)
        
        return feedback
    
    def close(self, feedback_id: int) -> Feedback:
        """关闭反馈"""
        feedback = Feedback.objects.get(id=feedback_id)
        feedback.status = 'closed'
        feedback.closed_at = now()
        feedback.save()
        
        return feedback

## 四、定时任务设计

### 4.1 催办提醒任务

```python
# 定时检查超期未审批任务
class ApprovalReminderTask:
    """审批超时提醒定时任务"""
    
    def run(self):
        # 查找超过48小时未审批的任务
        overdue_requests = ApprovalRequest.objects.filter(
            status='pending',
            submitted_at__lt=now() - timedelta(hours=48)
        )
        
        for request in overdue_requests:
            # 发送提醒给审批人
            self._send_reminder(request)
        
        # 查找超过24小时未审批的任务（可催办）
        reminder_eligible = ApprovalRequest.objects.filter(
            status='pending',
            submitted_at__lt=now() - timedelta(hours=24)
        )
        
        return {
            'overdue': overdue_requests.count(),
            'reminder_eligible': reminder_eligible.count()
        }
```

# 每日待办汇总
class DailyPendingSummaryTask:
    """每日待办汇总定时任务"""
    
    def run(self):
        # 每天上午9:00执行
        users = User.objects.filter(is_active=True)
        
        for user in users:
            pending_count = ApprovalRequest.objects.filter(
                current_approver=user,
                status='pending'
            ).count()
            
            if pending_count > 0:
                # 发送汇总消息
                MessageService.send_summary(user, pending_count)

## 六、安全设计

### 6.1 权限控制

| 功能 | 权限要求 |
|------|----------|
| 创建审批链 | 管理员 |

| 删除审批链 | 管理员 |

### 6.2 数据隔离

- 审批记录：只能查看自己提交的申请
- 待审批列表：只能看到分配给自己的审批
- 反馈列表：只能查看自己提交的反馈

1. 审批流程引擎正确处理或签和会签模式

5. 未读角标实时更新
6. 反馈状态按规则正确流转
7. API 响应时间符合性能要求
