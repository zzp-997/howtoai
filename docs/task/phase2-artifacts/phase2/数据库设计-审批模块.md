## 一、数据库表设计

### 1.1 审批链配置表 (approval_chains)

```sql
-- 审批链配置表
CREATE TABLE approval_chains (
    id BIGSERIAL PRIMARY KEY,
    business_type VARCHAR(50) NOT NULL,           -- 业务类型: trip/leave/reimburse/meeting
    name VARCHAR(100) NOT NULL,                   -- 审批链名称
    description TEXT,                             -- 描述
    is_enabled BOOLEAN DEFAULT true,              -- 是否启用
    created_by BIGINT NOT NULL,                   -- 创建人
    created_at TIMESTAMPTZ DEFAULT NOW(),
    updated_at TIMESTAMPTZ DEFAULT NOW()
);

-- 索引
CREATE INDEX idx_approval_chains_business_type ON approval_chains(business_type);
CREATE INDEX idx_approval_chains_is_enabled ON approval_chains(is_enabled);

-- 唯一约束
ALTER TABLE approval_chains ADD CONSTRAINT uk_approval_chains_business_type UNIQUE (business_type);
```

### 1.2 审批节点表 (approval_nodes)

```sql
-- 审批节点表
CREATE TABLE approval_nodes (
    id BIGSERIAL PRIMARY KEY,
    chain_id BIGINT NOT NULL REFERENCES approval_chains(id) ON DELETE CASCADE,
    node_order INTEGER NOT NULL,                 -- 节点顺序: 1, 2, 3...
    node_type VARCHAR(20) NOT NULL,              -- 节点类型: role/user/department_head
    node_value TEXT NOT NULL,                    -- 节点值: 角色ID/用户ID/部门ID
    approval_mode VARCHAR(10) NOT NULL,           -- 审批模式: or/and
    created_at TIMESTAMPTZ DEFAULT NOW()
);

-- 索引
CREATE INDEX idx_approval_nodes_chain_id ON approval_nodes(chain_id);
CREATE INDEX idx_approval_nodes_order ON approval_nodes(chain_id, node_order);

-- 外键约束
ALTER TABLE approval_nodes ADD CONSTRAINT fk_approval_nodes_chain 
    FOREIGN KEY (chain_id) REFERENCES approval_chains(id) ON DELETE CASCADE;
```

### 1.3 审批申请记录表 (approval_requests)

```sql
-- 审批申请记录表
CREATE TABLE approval_requests (
    id BIGSERIAL PRIMARY KEY,
    request_no VARCHAR(50) UNIQUE NOT NULL,       -- 申请单号
    business_type VARCHAR(50) NOT NULL,           -- 业务类型
    chain_id BIGINT REFERENCES approval_chains(id),
    applicant_id BIGINT NOT NULL,                 -- 申请人ID
    title VARCHAR(200) NOT NULL,                  -- 申请标题
    content JSONB,                                 -- 申请内容（JSON格式）
    attachments JSONB,                            -- 附件列表
    current_node_id BIGINT,                       -- 当前审批节点ID
    current_approver_id BIGINT,                   -- 当前审批人ID
    status VARCHAR(20) DEFAULT 'pending',          -- 状态: pending/approved/rejected
    submitted_at TIMESTAMPTZ DEFAULT NOW(),        -- 提交时间
    completed_at TIMESTAMPTZ,                      -- 完成时间
    created_at TIMESTAMPTZ DEFAULT NOW(),
    updated_at TIMESTAMPTZ DEFAULT NOW()
);

-- 索引
CREATE INDEX idx_approval_requests_request_no ON approval_requests(request_no);
CREATE INDEX idx_approval_requests_applicant_id ON approval_requests(applicant_id);
CREATE INDEX idx_approval_requests_status ON approval_requests(status);
CREATE INDEX idx_approval_requests_business_type ON approval_requests(business_type);
CREATE INDEX idx_approval_requests_current_approver ON approval_requests(current_approver_id, status);
```

### 1.4 审批记录表 (approval_records)

```sql
-- 审批记录表
CREATE TABLE approval_records (
    id BIGSERIAL PRIMARY KEY,
    request_id BIGINT NOT NULL REFERENCES approval_requests(id) ON DELETE CASCADE,
    node_id BIGINT NOT NULL REFERENCES approval_nodes(id),
    approver_id BIGINT NOT NULL,                  -- 审批人ID
    approver_name VARCHAR(100),                   -- 审批人姓名
    action VARCHAR(20) NOT NULL,                 -- 操作: approve/reject
    comment TEXT,                                 -- 审批意见
    created_at TIMESTAMPTZ DEFAULT NOW()
);

-- 索引
CREATE INDEX idx_approval_records_request_id ON approval_records(request_id);
CREATE INDEX idx_approval_records_approver_id ON approval_records(approver_id);
CREATE INDEX idx_approval_records_created_at ON approval_records(created_at);

-- 外键
ALTER TABLE approval_records ADD CONSTRAINT fk_approval_records_request 
    FOREIGN KEY (request_id) REFERENCES approval_requests(id) ON DELETE CASCADE;
ALTER TABLE approval_records ADD CONSTRAINT fk_approval_records_node 
    FOREIGN KEY (node_id) REFERENCES approval_nodes(id);
```

### 1.5 消息表 (messages)

```sql
-- 消息表
CREATE TABLE messages (
    id BIGSERIAL PRIMARY KEY,
    user_id BIGINT NOT NULL,                      -- 接收用户ID
    type VARCHAR(20) NOT NULL,                   -- 消息类型: approval/system/task
    title VARCHAR(200) NOT NULL,                  -- 消息标题
    content TEXT,                                 -- 消息内容
    related_type VARCHAR(50),                     -- 关联类型: approval_request/feedback
    related_id BIGINT,                            -- 关联ID
    is_read BOOLEAN DEFAULT false,                -- 是否已读
    read_at TIMESTAMPTZ,                          -- 阅读时间
    created_at TIMESTAMPTZ DEFAULT NOW()
);

-- 索引
CREATE INDEX idx_messages_user_id ON messages(user_id);
CREATE INDEX idx_messages_user_id_unread ON messages(user_id, is_read) WHERE is_read = false;
CREATE INDEX idx_messages_type ON messages(type);
CREATE INDEX idx_messages_created_at ON messages(created_at DESC);

-- 部分索引：未读消息
CREATE INDEX idx_messages_unread ON messages(user_id, created_at DESC) WHERE is_read = false;
```

### 1.6 意见反馈表 (feedbacks)

```sql
-- 意见反馈表
CREATE TABLE feedbacks (
    id BIGSERIAL PRIMARY KEY,
    feedback_no VARCHAR(50) UNIQUE NOT NULL,      -- 反馈单号
    user_id BIGINT NOT NULL,                       -- 提交用户ID
    user_name VARCHAR(100),                        -- 用户姓名
    type VARCHAR(20) NOT NULL,                    -- 反馈类型: suggestion/bug/optimization/other
    title VARCHAR(200) NOT NULL,                  -- 反馈标题
    content TEXT NOT NULL,                         -- 反馈内容
    images JSONB,                                  -- 图片列表
    status VARCHAR(20) DEFAULT 'pending',         -- 状态: pending/processing/replied/closed
    handler_id BIGINT,                             -- 处理人ID
    handler_name VARCHAR(100),                     -- 处理人姓名
    reply_content TEXT,                            -- 回复内容
    replied_at TIMESTAMPTZ,                       -- 回复时间
    closed_at TIMESTAMPTZ,                         -- 关闭时间
    created_at TIMESTAMPTZ DEFAULT NOW(),
    updated_at TIMESTAMPTZ DEFAULT NOW()
);

-- 索引
CREATE INDEX idx_feedbacks_feedback_no ON feedbacks(feedback_no);
CREATE INDEX idx_feedbacks_user_id ON feedbacks(user_id);
CREATE INDEX idx_feedbacks_status ON feedbacks(status);
CREATE INDEX idx_feedbacks_type ON feedbacks(type);
CREATE INDEX idx_feedbacks_created_at ON feedbacks(created_at DESC);
```

### 1.7 催办记录表 (approval_reminders)

```sql
-- 催办记录表
CREATE TABLE approval_reminders (
    id BIGSERIAL PRIMARY KEY,
    request_id BIGINT NOT NULL REFERENCES approval_requests(id) ON DELETE CASCADE,
    requester_id BIGINT NOT NULL,                  -- 催办人ID
    reminder_count INTEGER DEFAULT 1,              -- 催办次数
    last_reminded_at TIMESTAMPTZ DEFAULT NOW(),
    created_at TIMESTAMPTZ DEFAULT NOW()
);

-- 索引
CREATE INDEX idx_approval_reminders_request_id ON approval_reminders(request_id);
CREATE UNIQUE INDEX idx_approval_reminders_request_requester ON approval_reminders(request_id, requester_id);
```

## 三、SQL迁移脚本

### 3.1 创建表结构

```sql
-- 执行顺序: 按依赖关系依次执行

-- 1. 审批链配置表
CREATE TABLE approval_chains (
    id BIGSERIAL PRIMARY KEY,
    business_type VARCHAR(50) NOT NULL,
    name VARCHAR(100) NOT NULL,
    description TEXT,
    is_enabled BOOLEAN DEFAULT true,
    created_by BIGINT NOT NULL,
    created_at TIMESTAMPTZ DEFAULT NOW(),
    updated_at TIMESTAMPTZ DEFAULT NOW()

CREATE INDEX idx_approval_chains_business_type ON approval_chains(business_type);

ALTER TABLE approval_chains ADD CONSTRAINT uk_approval_chains_business_type UNIQUE (business_type);

-- 2. 审批节点表
CREATE TABLE approval_nodes (
    id BIGSERIAL PRIMARY KEY,
    chain_id BIGINT NOT NULL,
    node_order INTEGER NOT NULL,
    node_type VARCHAR(20) NOT NULL,
    node_value TEXT NOT NULL,
    approval_mode VARCHAR(10) NOT NULL,
    created_at TIMESTAMPTZ DEFAULT NOW(),
    CONSTRAINT fk_approval_nodes_chain FOREIGN KEY (chain_id) REFERENCES approval_chains(id) ON DELETE CASCADE
);

CREATE INDEX idx_approval_nodes_chain_id ON approval_nodes(chain_id);
CREATE INDEX idx_approval_nodes_order ON approval_nodes(chain_id, node_order);

-- 3. 审批申请记录表
CREATE TABLE approval_requests (
    id BIGSERIAL PRIMARY KEY,
    request_no VARCHAR(50) UNIQUE NOT NULL,
    business_type VARCHAR(50) NOT NULL,
    chain_id BIGINT,
    applicant_id BIGINT NOT NULL,
    title VARCHAR(200) NOT NULL,
    content JSONB,
    attachments JSONB,
    current_node_id BIGINT,
    current_approver_id BIGINT,
    status VARCHAR(20) DEFAULT 'pending',
    submitted_at TIMESTAMPTZ DEFAULT NOW(),
    completed_at TIMESTAMPTZ,
    created_at TIMESTAMPTZ DEFAULT NOW(),
    updated_at TIMESTAMPTZ DEFAULT NOW()
);

CREATE INDEX idx_approval_requests_request_no ON approval_requests(request_no);
CREATE INDEX idx_approval_requests_applicant_id ON approval_requests(applicant_id);
CREATE INDEX idx_approval_requests_status ON approval_requests(status);
CREATE INDEX idx_approval_requests_business_type ON approval_requests(business_type);
CREATE INDEX idx_approval_requests_current_approver ON approval_requests(current_approver_id, status);

-- 4. 审批记录表
CREATE TABLE approval_records (
    id BIGSERIAL PRIMARY KEY,
    request_id BIGINT NOT NULL,
    node_id BIGINT NOT NULL,
    approver_id BIGINT NOT NULL,

    action VARCHAR(20) NOT NULL,

    CONSTRAINT fk_approval_records_request FOREIGN KEY (request_id) REFERENCES approval_requests(id) ON DELETE CASCADE,
    CONSTRAINT fk_approval_records_node FOREIGN KEY (node_id) REFERENCES approval_nodes(id)
);

CREATE INDEX idx_approval_records_request_id ON approval_records(request_id);
CREATE INDEX idx_approval_records_approver_id ON approval_records(approver_id);
CREATE INDEX idx_approval_records_created_at ON approval_records(created_at);

-- 5. 消息表
CREATE TABLE messages (
    id BIGSERIAL PRIMARY KEY,
    user_id BIGINT NOT NULL,
    type VARCHAR(20) NOT NULL,
    title VARCHAR(200) NOT NULL,
    content TEXT,
    related_type VARCHAR(50),
    related_id BIGINT,
    is_read BOOLEAN DEFAULT false,
    read_at TIMESTAMPTZ,
    created_at TIMESTAMPTZ DEFAULT NOW()
);

CREATE INDEX idx_messages_user_id ON messages(user_id);
CREATE INDEX idx_messages_user_id_unread ON messages(user_id, is_read) WHERE is_read = false;
CREATE INDEX idx_messages_type ON messages(type);

-- 6. 意见反馈表
CREATE TABLE feedbacks (
    id BIGSERIAL PRIMARY KEY,
    feedback_no VARCHAR(50) UNIQUE NOT NULL,
    user_id BIGINT NOT NULL,
    user_name VARCHAR(100),
    type VARCHAR(20) NOT NULL,
    title VARCHAR(200) NOT NULL,
    content TEXT NOT NULL,
    images JSONB,
    status VARCHAR(20) DEFAULT 'pending',
    handler_id BIGINT,
    handler_name VARCHAR(100),
    reply_content TEXT,
    replied_at TIMESTAMPTZ,
    closed_at TIMESTAMPTZ,
    created_at TIMESTAMPTZ DEFAULT NOW(),
    updated_at TIMESTAMPTZ DEFAULT NOW()
);

CREATE INDEX idx_feedbacks_feedback_no ON feedbacks(feedback_no);
CREATE INDEX idx_feedbacks_user_id ON feedbacks(user_id);
CREATE INDEX idx_feedbacks_status ON feedbacks(status);
CREATE INDEX idx_feedbacks_type ON feedbacks(type);
CREATE INDEX idx_feedbacks_created_at ON feedbacks(created_at DESC);

-- 7. 催办记录表
CREATE TABLE approval_reminders (
    id BIGSERIAL PRIMARY KEY,
    request_id BIGINT NOT NULL,
    requester_id BIGINT NOT NULL,
    reminder_count INTEGER DEFAULT 1,
    last_reminded_at TIMESTAMPTZ DEFAULT NOW(),
    created_at TIMESTAMPTZ DEFAULT NOW(),
    CONSTRAINT fk_approval_reminders_request FOREIGN KEY (request_id) REFERENCES approval_requests(id) ON DELETE CASCADE

CREATE INDEX idx_approval_reminders_request_id ON approval_reminders(request_id);
CREATE UNIQUE INDEX idx_approval_reminders_request_requester ON approval_reminders(request_id, requester_id);
```

## 四、性能优化建议

### 4.1 索引优化

- 审批列表查询：复合索引 (status, submitted_at)
- 待办审批：复合索引 (current_approver_id, status)
- 消息未读计数：部分索引

### 4.2 分区策略

- 审批记录表按月份分区
- 消息表按月份分区

### 4.3 查询优化

- 避免 SELECT *，只查询必要字段
- 使用游标分页替代 OFFSET
- 批量操作使用 INSERT ... ON CONFLICT
