```vue
<script setup>
import { ref, reactive, onMounted } from 'vue'
import { showToast, showConfirmDialog } from 'vant'
import { getChains, createChain, updateChain, deleteChain } from '@/api/approval'

const activeTab = ref('all')
const chains = ref([])
const loading = ref(false)
const showPopup = ref(false)
const isEdit = ref(false)
const currentId = ref(null)
const showDeleteConfirm = ref(false)
const deleteTarget = ref(null)

const formData = reactive({
  business_type: '',
  name: '',
  description: '',
  is_enabled: true,
  nodes: []
})

const businessTypes = [
  { text: '差旅申请', value: 'trip' },
  { text: '请假申请', value: 'leave' },
  { text: '报销申请', value: 'reimburse' },
  { text: '会议审批', value: 'meeting' }
]

const nodeTypes = [
  { text: '按角色', value: 'role' },
  { text: '指定用户', value: 'user' },
  { text: '部门负责人', value: 'department_head' }
]

const approvalModes = [
  { text: '或签', value: 'or' },
  { text: '会签', value: 'and' }
]

// 获取审批链列表
const fetchChains = async () => {
  loading.value = true
  try {
    const params = activeTab.value !== 'all' 
      ? { business_type: activeTab.value }
      : {}
    const { data } = await getChains(params)
    chains.value = data || []
  } catch (error) {
    showToast('获取列表失败')
  } finally {
    loading.value = false
  }
}

const handleTabChange = () => {
  fetchChains()
}

const getNodeLabel = (node) => {
  if (node.node_type === 'role') {
    return `角色: ${node.node_value}`
  } else if (node.node_type === 'user') {
    return `用户: ${node.node_value}`
  } else {
    return '部门负责人'
  }
}

const handleAdd = () => {
  isEdit.value = false
  currentId.value = null
  Object.assign(formData, {
    business_type: '',
    name: '',
    description: '',
    is_enabled: true,
    nodes: [{ node_order: 1, node_type: 'role', node_value: '', approval_mode: 'or' }]
  })
  showPopup.value = true
}

const handleEdit = (chain) => {
  isEdit.value = true
  currentId.value = chain.id
  Object.assign(formData, {
    business_type: chain.business_type,
    name: chain.name,
    description: chain.description,
    is_enabled: chain.is_enabled,
    nodes: chain.nodes.map(n => ({
      node_order: n.node_order,
      node_type: n.node_type,
      node_value: n.node_value,
      approval_mode: n.approval_mode
    }))
  })
  showPopup.value = true
}

const handleDelete = (chain) => {
  deleteTarget.value = chain
  showDeleteConfirm.value = true
}

const confirmDelete = async () => {
  try {
    await deleteChain(deleteTarget.value.id)
    showToast('删除成功')
    fetchChains()
  } catch (error) {
    showToast('删除失败')
  }
}

const addNode = () => {
  formData.nodes.push({
    node_order: formData.nodes.length + 1,
    node_type: 'role',
    node_value: '',
    approval_mode: 'or'
  })
}

const removeNode = (index) => {
  formData.nodes.splice(index, 1)
  // 重新排序
  formData.nodes.forEach((node, i) => {
    node.node_order = i + 1
  })
}

const handleSubmit = async () => {
  try {
    if (isEdit.value) {
      await updateChain(currentId.value, formData)
      showToast('更新成功')
    } else {
      await createChain(formData)
      showToast('创建成功')
    }
    showPopup.value = false
    fetchChains()
  } catch (error) {
    showToast(error.message || '操作失败')
  }
}

onMounted(() => {
  fetchChains()
})
</script>
```

### 3.2 进度节点图组件 (ProgressGraph.vue)

```vue
<template>
  <div class="progress-graph">
    <div class="nodes-container">
      <div
        v-for="(node, index) in nodes"
        :key="node.id"
        class="node-wrapper"
      >
        <!-- 节点 -->
        <div
          class="node"
          :class="getNodeClass(node)"
        >
          <van-icon :name="getNodeIcon(node)" />
        </div>
        
        <!-- 节点标签 -->
        <div class="node-label" :class="{ active: isNodeActive(node) }">
          {{ getNodeLabel(node) }}
        </div>

        <!-- 连接线 -->
        <div
          v-if="index < nodes.length - 1"
          class="connector"
          :class="{ completed: isConnectorCompleted(node, nodes[index + 1]) }"
        />
      </div>
    </div>

    <!-- 状态说明 -->
    <div class="legend">
      <div class="legend-item">
        <span class="legend-dot completed" />
        <span>已完成</span>
      </div>
      <div class="legend-item">
        <span class="legend-dot active" />
        <span>进行中</span>
      </div>
      <div class="legend-item">
        <span class="legend-dot pending" />
        <span>待处理</span>
      </div>
    </div>
  </div>
</template>

<script setup>
const props = defineProps({
  nodes: {
    type: Array,
    default: () => []
  },
  records: {
    type: Array,
    default: () => []
  },
  currentNodeId: {
    type: Number,
    default: null
  },
  status: {
    type: String,
    default: 'pending'
  }
})

const getNodeClass = (node) => {
  if (props.status === 'rejected') {
    const nodeIndex = props.nodes.findIndex(n => n.id === node.id)
    const rejectedIndex = props.nodes.findIndex(n => n.id === props.currentNodeId)
    if (nodeIndex < rejectedIndex) return 'completed'
    if (nodeIndex === rejectedIndex) return 'rejected'
    return 'pending'
  }

  if (props.status === 'approved') {
    return 'completed'
  }

  if (node.id === props.currentNodeId) {
    return 'active'
  }

  const nodeRecord = props.records.find(r => r.node_id === node.id)
  if (nodeRecord) return 'completed'

  return 'pending'
}

const getNodeIcon = (node) => {
  const nodeClass = getNodeClass(node)
  if (nodeClass === 'completed') return 'success'
  if (nodeClass === 'active') return 'clock'
  if (nodeClass === 'rejected') return 'cross'
  return 'circle'
}

const getNodeLabel = (node) => {
  if (node.node_type === 'role') {
    return node.node_value
  } else if (node.node_type === 'user') {
    return '指定用户'
  } else {
    return '部门负责人'
  }
}

const isNodeActive = (node) => {
  return node.id === props.currentNodeId
}

const isConnectorCompleted = (currentNode, nextNode) => {
  const currentRecord = props.records.find(r => r.node_id === currentNode.id)
  return !!currentRecord
}
</script>
```

```vue
  selectedIds: {
    type: Array,
    default: () => []
  },

    type: Number,
    default: 20
  }

const confirmType = ref('')
const confirmTitle = ref('')

const handleBatchApprove = () => {

    return
  }
  confirmType.value = '批量通过'
  confirmTitle.value = '批量审批确认'
  actionType.value = 'approve'
  showConfirm.value = true
}

const handleBatchReject = () => {
  if (props.selectedCount > props.maxCount) {
    showToast(`单次最多批量审批 ${props.maxCount} 项`)

  actionType.value = 'reject'

    comment: comment.value
```
