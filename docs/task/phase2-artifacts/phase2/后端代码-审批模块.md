```python
from datetime import datetime
from sqlalchemy import Column, Integer, String, Boolean, Text, DateTime, ForeignKey, JSON
from sqlalchemy.orm import relationship
from sqlalchemy.ext.declarative import declarative_base

Base = declarative_base()


class ApprovalChain(Base):
    """审批链配置表"""
    __tablename__ = 'approval_chains'

    id = Column(Integer, primary_key=True, autoincrement=True)
    business_type = Column(String(50), nullable=False, unique=True)  # 业务类型
    name = Column(String(100), nullable=False)  # 审批链名称
    description = Column(Text)  # 描述
    is_enabled = Column(Boolean, default=True)  # 是否启用
    created_by = Column(Integer, nullable=False)  # 创建人
    created_at = Column(DateTime, default=datetime.utcnow)
    updated_at = Column(DateTime, default=datetime.utcnow, onupdate=datetime.utcnow)

    # 关联
    nodes = relationship('ApprovalNode', back_populates='chain', cascade='all, delete-orphan')
    requests = relationship('ApprovalRequest', back_populates='chain')

    def to_dict(self):
        return {
            'id': self.id,
            'business_type': self.business_type,
            'name': self.name,
            'description': self.description,
            'is_enabled': self.is_enabled,
            'created_by': self.created_by,
            'created_at': self.created_at.isoformat() if self.created_at else None,
            'updated_at': self.updated_at.isoformat() if self.updated_at else None,
            'nodes': [node.to_dict() for node in self.nodes]
        }


class ApprovalNode(Base):
    """审批节点表"""
    __tablename__ = 'approval_nodes'

    id = Column(Integer, primary_key=True, autoincrement=True)
    chain_id = Column(Integer, ForeignKey('approval_chains.id', ondelete='CASCADE'), nullable=False)
    node_order = Column(Integer, nullable=False)  # 节点顺序
    node_type = Column(String(20), nullable=False)  # 节点类型: role/user/department_head
    node_value = Column(Text, nullable=False)  # 节点值
    approval_mode = Column(String(10), nullable=False)  # 审批模式: or/and
    created_at = Column(DateTime, default=datetime.utcnow)

    # 关联
    chain = relationship('ApprovalChain', back_populates='nodes')

    def to_dict(self):
        return {
            'id': self.id,
            'chain_id': self.chain_id,
            'node_order': self.node_order,
            'node_type': self.node_type,
            'node_value': self.node_value,
            'approval_mode': self.approval_mode,
            'created_at': self.created_at.isoformat() if self.created_at else None
        }


class ApprovalRequest(Base):
    """审批申请记录表"""
    __tablename__ = 'approval_requests'

    id = Column(Integer, primary_key=True, autoincrement=True)
    request_no = Column(String(50), unique=True, nullable=False)  # 申请单号
    business_type = Column(String(50), nullable=False)  # 业务类型
    chain_id = Column(Integer, ForeignKey('approval_chains.id'))
    applicant_id = Column(Integer, nullable=False)  # 申请人ID
    title = Column(String(200), nullable=False)  # 申请标题
    content = Column(JSON)  # 申请内容
    attachments = Column(JSON)  # 附件
    current_node_id = Column(Integer, ForeignKey('approval_nodes.id'))
    current_approver_id = Column(Integer)  # 当前审批人ID
    status = Column(String(20), default='pending')  # pending/approved/rejected
    submitted_at = Column(DateTime, default=datetime.utcnow)
    completed_at = Column(DateTime)
    created_at = Column(DateTime, default=datetime.utcnow)
    updated_at = Column(DateTime, default=datetime.utcnow, onupdate=datetime.utcnow)

    # 关联
    chain = relationship('ApprovalChain', back_populates='requests')
    current_node = relationship('ApprovalNode')
    records = relationship('ApprovalRecord', back_populates='request', cascade='all, delete-orphan')
    reminders = relationship('ApprovalReminder', back_populates='request', cascade='all, delete-orphan')

    def to_dict(self):
        return {
            'id': self.id,
            'request_no': self.request_no,
            'business_type': self.business_type,
            'chain_id': self.chain_id,
            'applicant_id': self.applicant_id,
            'title': self.title,
            'content': self.content,
            'attachments': self.attachments,
            'current_node_id': self.current_node_id,
            'current_approver_id': self.current_approver_id,
            'status': self.status,
            'submitted_at': self.submitted_at.isoformat() if self.submitted_at else None,
            'completed_at': self.completed_at.isoformat() if self.completed_at else None,
            'created_at': self.created_at.isoformat() if self.created_at else None,
            'updated_at': self.updated_at.isoformat() if self.updated_at else None
        }


class ApprovalRecord(Base):
    """审批记录表"""
    __tablename__ = 'approval_records'

    id = Column(Integer, primary_key=True, autoincrement=True)
    request_id = Column(Integer, ForeignKey('approval_requests.id', ondelete='CASCADE'), nullable=False)
    node_id = Column(Integer, ForeignKey('approval_nodes.id'), nullable=False)
    approver_id = Column(Integer, nullable=False)  # 审批人ID
    approver_name = Column(String(100))  # 审批人姓名
    action = Column(String(20), nullable=False)  # approve/reject
    comment = Column(Text)  # 审批意见
    created_at = Column(DateTime, default=datetime.utcnow)

    # 关联
    request = relationship('ApprovalRequest', back_populates='records')
    node = relationship('ApprovalNode')

    def to_dict(self):
        return {
            'id': self.id,
            'request_id': self.request_id,
            'node_id': self.node_id,
            'approver_id': self.approver_id,
            'approver_name': self.approver_name,
            'action': self.action,
            'comment': self.comment,
            'created_at': self.created_at.isoformat() if self.created_at else None
        }


class ApprovalReminder(Base):
    """催办记录表"""
    __tablename__ = 'approval_reminders'

    id = Column(Integer, primary_key=True, autoincrement=True)
    request_id = Column(Integer, ForeignKey('approval_requests.id', ondelete='CASCADE'), nullable=False)
    requester_id = Column(Integer, nullable=False)  # 催办人ID
    reminder_count = Column(Integer, default=1)  # 催办次数
    last_reminded_at = Column(DateTime, default=datetime.utcnow)
    created_at = Column(DateTime, default=datetime.utcnow)

    # 关联
    request = relationship('ApprovalRequest', back_populates='reminders')

    def to_dict(self):
        return {
            'id': self.id,
            'request_id': self.request_id,
            'requester_id': self.requester_id,
            'reminder_count': self.reminder_count,
            'last_reminded_at': self.last_reminded_at.isoformat() if self.last_reminded_at else None,
            'created_at': self.created_at.isoformat() if self.created_at else None
        }
```

```python
from flask import Blueprint, request, jsonify
from flask_jwt_extended import jwt_required, get_jwt_identity
from app.services.approval_service import ApprovalService
from app.schemas.approval import ChainSchema, NodeSchema

approval_bp = Blueprint('approval', __name__, url_prefix='/api/approval')
chain_schema = ChainSchema()
node_schema = NodeSchema()
approval_service = ApprovalService()


@approval_bp.route('/chains', methods=['GET'])
@jwt_required()
def get_chains():
    """获取审批链列表"""
    business_type = request.args.get('business_type')
    is_enabled = request.args.get('is_enabled')

    chains = approval_service.list_chains(
        business_type=business_type,
        is_enabled=is_enabled
    )

    return jsonify({
        'success': True,
        'data': [chain.to_dict() for chain in chains]
    })


@approval_bp.route('/chains/<int:chain_id>', methods=['GET'])
@jwt_required()
def get_chain(chain_id):
    """获取审批链详情"""
    chain = approval_service.get_chain(chain_id)

    if not chain:
        return jsonify({'success': False, 'error': '审批链不存在'}), 404

    return jsonify({
        'success': True,
        'data': chain.to_dict()
    })


@approval_bp.route('/chains', methods=['POST'])
@jwt_required()
def create_chain():
    """创建审批链"""
    current_user_id = get_jwt_identity()
    data = request.get_json()

    # 验证数据
    errors = chain_schema.validate(data)
    if errors:
        return jsonify({'success': False, 'error': errors}), 400

    chain = approval_service.create_chain(data, current_user_id)

    return jsonify({
        'success': True,
        'data': chain.to_dict()
    }), 201


@approval_bp.route('/chains/<int:chain_id>', methods=['PUT'])
@jwt_required()
def update_chain(chain_id):
    """更新审批链"""
    data = request.get_json()

    errors = chain_schema.validate(data)
    if errors:
        return jsonify({'success': False, 'error': errors}), 400

    chain = approval_service.update_chain(chain_id, data)

    if not chain:
        return jsonify({'success': False, 'error': '审批链不存在'}), 404

    return jsonify({
        'success': True,
        'data': chain.to_dict()
    })


@approval_bp.route('/chains/<int:chain_id>', methods=['DELETE'])
@jwt_required()
def delete_chain(chain_id):
    """删除审批链"""
    success = approval_service.delete_chain(chain_id)

    if not success:
        return jsonify({'success': False, 'error': '审批链不存在'}), 404

    return jsonify({
        'success': True,
        'message': '删除成功'
    })
```

```python
from datetime import datetime, timedelta
from sqlalchemy.orm import Session
from app.models.approval import ApprovalChain, ApprovalNode, ApprovalRequest, ApprovalRecord, ApprovalReminder
from app.models.message import Message
from app.utils.generate_no import generate_request_no


class ApprovalService:
    def __init__(self, db: Session):
        self.db = db

    def list_chains(self, business_type=None, is_enabled=None):
        """获取审批链列表"""
        query = self.db.query(ApprovalChain)

        if business_type:
            query = query.filter(ApprovalChain.business_type == business_type)
        if is_enabled is not None:
            query = query.filter(ApprovalChain.is_enabled == (is_enabled == 'true'))

        return query.order_by(ApprovalChain.id).all()

    def get_chain(self, chain_id):
        """获取审批链详情"""
        return self.db.query(ApprovalChain).filter(ApprovalChain.id == chain_id).first()

    def create_chain(self, data, created_by):
        """创建审批链"""
        chain = ApprovalChain(
            business_type=data['business_type'],
            name=data['name'],
            description=data.get('description'),
            is_enabled=data.get('is_enabled', True),
            created_by=created_by
        )
        self.db.add(chain)
        self.db.flush()

        # 创建节点
        for node_data in data.get('nodes', []):
            node = ApprovalNode(
                chain_id=chain.id,
                node_order=node_data['node_order'],
                node_type=node_data['node_type'],
                node_value=str(node_data['node_value']),
                approval_mode=node_data['approval_mode']
            )
            self.db.add(node)

        self.db.commit()
        self.db.refresh(chain)
        return chain

    def update_chain(self, chain_id, data):
        """更新审批链"""
        chain = self.get_chain(chain_id)
        if not chain:
            return None

        if 'name' in data:
            chain.name = data['name']
        if 'description' in data:
            chain.description = data['description']
        if 'is_enabled' in data:
            chain.is_enabled = data['is_enabled']

        # 更新节点
        if 'nodes' in data:
            # 删除旧节点
            self.db.query(ApprovalNode).filter(ApprovalNode.chain_id == chain_id).delete()

            # 创建新节点
            for node_data in data['nodes']:
                node = ApprovalNode(
                    chain_id=chain.id,
                    node_order=node_data['node_order'],
                    node_type=node_data['node_type'],
                    node_value=str(node_data['node_value']),
                    approval_mode=node_data['approval_mode']
                )
                self.db.add(node)

        self.db.commit()
        self.db.refresh(chain)
        return chain

    def delete_chain(self, chain_id):
        """删除审批链"""
        chain = self.get_chain(chain_id)
        if not chain:
            return False

        self.db.delete(chain)
        self.db.commit()
        return True

    def submit_request(self, data, applicant_id):
        """提交审批申请"""
        # 获取审批链
        chain = self.db.query(ApprovalChain).filter(
            ApprovalChain.business_type == data['business_type'],
            ApprovalChain.is_enabled == True
        ).first()

        if not chain:
            raise ValueError('该业务类型未配置审批流程')

        # 创建申请
        request = ApprovalRequest(
            request_no=generate_request_no('REQ'),
            business_type=data['business_type'],
            chain_id=chain.id,
            applicant_id=applicant_id,
            title=data['title'],
            content=data.get('content'),
            attachments=data.get('attachments'),
            status='pending'

        self.db.add(request)
        self.db.flush()

        # 设置第一个审批节点

            ApprovalNode.chain_id == chain.id
        ).order_by(ApprovalNode.node_order).first()

        if first_node:
            request.current_node_id = first_node.id
            request.current_approver_id = self._resolve_approver(first_node, applicant_id)

        self.db.commit()
        self.db.refresh(request)

        # 发送通知
        self._send_notification(request, 'submitted')

        return request

    def approve(self, request_id, approver_id, comment):
        """审批通过"""
        approval_request = self.db.query(ApprovalRequest).filter(
            ApprovalRequest.id == request_id
        ).first()

        if not approval_request:
            return {'success': False, 'error': '申请不存在'}

        if approval_request.current_approver_id != approver_id:
            return {'success': False, 'error': '您不是当前审批人'}

        # 记录审批
        record = ApprovalRecord(
            request_id=request_id,
            node_id=approval_request.current_node_id,
            approver_id=approver_id,
            approver_name='',  # 从用户表获取
            action='approve',
            comment=comment
        )
        self.db.add(record)

        # 处理下一节点
        next_node = self._move_to_next_node(approval_request)

        if next_node:

            approval_request.current_approver_id = self._resolve_approver(next_node, approval_request.applicant_id)
            self._send_notification(approval_request, 'pending')
        else:
            approval_request.status = 'approved'
            approval_request.completed_at = datetime.utcnow()
            self._send_notification(approval_request, 'approved')

        self.db.commit()
        self.db.refresh(approval_request)

        return {'success': True, 'data': approval_request.to_dict()}

            approver_id=approver_id,
            approver_name='',
            action='reject',
            comment=comment

        self.db.add(record)

        approval_request.status = 'rejected'
        approval_request.completed_at = datetime.utcnow()

        self.db.commit()
        self.db.refresh(approval_request)

        # 发送通知
        self._send_notification(approval_request, 'rejected')

        return {'success': True, 'data': approval_request.to_dict()}

    def batch_approve(self, request_ids, approver_id, comment):
        """批量审批通过"""
        results = {'success': [], 'failed': []}

        for req_id in request_ids:
            result = self.approve(req_id, approver_id, comment)
            if result['success']:
                results['success'].append(req_id)
            else:
                results['failed'].append({'id': req_id, 'error': result['error']})

        return {'success': True, 'data': results}

    def batch_reject(self, request_ids, approver_id, comment):
        """批量审批拒绝"""
        results = {'success': [], 'failed': []}

        for req_id in request_ids:
            result = self.reject(req_id, approver_id, comment)
            if result['success']:
                results['success'].append(req_id)
            else:
                results['failed'].append({'id': req_id, 'error': result['error']})

        return {'success': True, 'data': results}

    def get_pending_list(self, user_id, status='pending'):
        """获取待审批列表"""
        return self.db.query(ApprovalRequest).filter(
            ApprovalRequest.current_approver_id == user_id,
            ApprovalRequest.status == status
        ).all()

    def get_progress(self, request_id, user_id):
        """获取审批进度"""
        request = self.db.query(ApprovalRequest).filter(
            ApprovalRequest.id == request_id
        ).first()

        if not request:
            return None

        # 权限检查
        if request.applicant_id != user_id:
            return None

        # 获取审批记录
        records = self.db.query(ApprovalRecord).filter(
            ApprovalRecord.request_id == request_id
        ).order_by(ApprovalRecord.created_at).all()

        # 获取节点信息
        nodes = self.db.query(ApprovalNode).filter(
            ApprovalNode.chain_id == request.chain_id
        ).order_by(ApprovalNode.node_order).all()

        return {
            'request': request.to_dict(),
            'nodes': [node.to_dict() for node in nodes],
            'records': [record.to_dict() for record in records]
        }

    def remind(self, request_id, requester_id):
        """催办"""
        request = self.db.query(ApprovalRequest).filter(
            ApprovalRequest.id == request_id
        ).first()

        if not request:
            return {'success': False, 'error': '申请不存在'}

        # 检查是否超过24小时
        if datetime.utcnow() - request.submitted_at < timedelta(hours=24):
            return {'success': False, 'error': '提交不足24小时，无法催办'}

        # 检查催办次数
        reminder = self.db.query(ApprovalReminder).filter(
            ApprovalReminder.request_id == request_id,
            ApprovalReminder.requester_id == requester_id
        ).first()

        if reminder and reminder.reminder_count >= 3:
            return {'success': False, 'error': '催办次数已达上限（3次）'}

        # 检查催办间隔
        if reminder and datetime.utcnow() - reminder.last_reminded_at < timedelta(hours=4):
            return {'success': False, 'error': '催办间隔需大于4小时'}

        # 更新或创建催办记录

            reminder.reminder_count += 1
            reminder.last_reminded_at = datetime.utcnow()
        else:
            reminder = ApprovalReminder(
                request_id=request_id,
                requester_id=requester_id,
                reminder_count=1
            )
            self.db.add(reminder)

        self.db.commit()

        # 发送催办通知
        self._send_notification(request, 'reminded')

        return {'success': True, 'message': '催办成功'}
```

```python
from marshmallow import Schema, fields, validate, validates, ValidationError


class NodeSchema(Schema):
    node_order = fields.Int(required=True, validate=validate.Range(min=1))
    node_type = fields.Str(required=True, validate=validate.OneOf(['role', 'user', 'department_head']))
    node_value = fields.Raw(required=True)
    approval_mode = fields.Str(required=True, validate=validate.OneOf(['or', 'and']))

    @validates('node_value')
    def validate_node_value(self, value):
        if isinstance(value, (list, dict)):
            return  # JSON类型值
        if not value or not str(value).strip():
            raise ValidationError('node_value不能为空')


class ChainSchema(Schema):
    business_type = fields.Str(required=True, validate=validate.OneOf(
        ['trip', 'leave', 'reimburse', 'meeting']
    ))
    name = fields.Str(required=True, validate=validate.Length(min=1, max=100))
    description = fields.Str(allow_none=True)
    is_enabled = fields.Bool(load_default=True)
    nodes = fields.List(fields.Nested(NodeSchema), validate=validate.Length(min=1))

    @validates('nodes')
    def validate_nodes(self, value):
        if not value:
            raise ValidationError('至少需要添加一个审批节点')

        # 检查顺序连续性
        orders = [node.get('node_order') for node in value if isinstance(node, dict)]
        orders.sort()
        for i, order in enumerate(orders, 1):
            if order != i:
                raise ValidationError(f'节点顺序必须连续，当前缺少节点{i}')


class SubmitRequestSchema(Schema):
    business_type = fields.Str(required=True, validate=validate.OneOf(
        ['trip', 'leave', 'reimburse', 'meeting']
    ))
    title = fields.Str(required=True, validate=validate.Length(min=1, max=200))
    content = fields.Dict(allow_none=True)
    attachments = fields.List(fields.Dict(), load_default=[])


class ApprovalActionSchema(Schema):
    comment = fields.Str(allow_none=True, validate=validate.Length(max=500))


class BatchActionSchema(Schema):
    request_ids = fields.List(fields.Int(), required=True, validate=validate.Length(min=1, max=20))
    comment = fields.Str(allow_none=True, validate=validate.Length(max=500))
```
