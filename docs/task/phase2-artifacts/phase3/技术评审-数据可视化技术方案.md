## 三、核心算法设计

### 3.1 会议室智能推荐算法

```python
class MeetingRoomRecommender:
    """会议室智能推荐器"""

    def recommend(self, criteria: dict) -> List[dict]:
        """
        推荐算法主流程
        1. 基础筛选：容量、设备
        2. 可用性计算：时间段匹配
        3. 评分排序：多维度评分
        """
        # Step 1: 容量筛选
        rooms = self._filter_by_capacity(criteria['attendees_count'])

        # Step 2: 设备筛选
        if criteria.get('required_equipment'):
            rooms = self._filter_by_equipment(rooms, criteria['required_equipment'])

        # Step 3: 可用性计算
        rooms = self._filter_available(rooms, criteria['date'], criteria['duration'])

        # Step 4: 评分排序
        scored_rooms = [self._score_room(room, criteria) for room in rooms]
        return sorted(scored_rooms, key=lambda x: x['score'], reverse=True)

    def _score_room(self, room, criteria):
        """计算会议室评分"""
        score = 100

        # 容量匹配度（-10 ~ +10）
        capacity_diff = room.capacity - criteria['attendees_count']
        score += min(10, capacity_diff) if capacity_diff > 0 else max(-10, capacity_diff)

        # 距离评分（-20 ~ +10）
        score += self._calculate_distance_score(room, criteria.get('preferred_floor'))

        # 利用率评分（-10 ~ +10）
        score += self._calculate_usage_score(room)

        return {
            'room': room,
            'score': score,
            'reasons': self._generate_reasons(room, score)
        }
```

### 3.2 冲突检测算法

```python
def check_booking_conflict(room_id, start_time, end_time):
    """检测会议室预订冲突"""
    existing = db.session.query(Meeting).filter(
        Meeting.room_id == room_id,
        Meeting.status.in_(['scheduled', 'in_progress']),
        Meeting.start_time < end_time,
        Meeting.end_time > start_time
    ).first()

    if existing:
        return {
            'has_conflict': True,
            'conflict_with': {
                'title': existing.title,
                'organizer': existing.organizer.name,
                'time': f"{existing.start_time} - {existing.end_time}"
            }
        }

    return {'has_conflict': False}
```

### 3.3 数据统计聚合算法

```python
def calculate_meeting_usage_rate(room_id, start_date, end_date):
    """计算会议室使用率"""
    total_minutes = db.session.query(
        func.sum(
            func.timedelta_minutes(Meeting.end_time - Meeting.start_time)
        )
    ).filter(
        Meeting.room_id == room_id,
        Meeting.start_time >= start_date,
        Meeting.end_time <= end_date,
        Meeting.status.in_(['scheduled', 'completed'])
    ).scalar() or 0

    # 统计周期内的可用总分钟数（假设每天8小时）
    days = (end_date - start_date).days + 1
    available_minutes = days * 8 * 60

    return round(total_minutes / available_minutes * 100, 2)
```
