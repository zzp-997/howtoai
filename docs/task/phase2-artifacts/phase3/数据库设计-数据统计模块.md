## 二、表结构设计

### 2.1 会议室基础信息表

```sql
CREATE TABLE meeting_rooms (
    id BIGINT PRIMARY KEY AUTO_INCREMENT COMMENT '会议室ID',
    name VARCHAR(100) NOT NULL COMMENT '会议室名称',
    floor VARCHAR(50) COMMENT '楼层',
    building VARCHAR(50) COMMENT '楼栋',
    capacity INT DEFAULT 0 COMMENT '容纳人数',
    equipment JSON COMMENT '设备列表["projector", "whiteboard", "video_conference"]',
    room_type ENUM('small', 'medium', 'large', 'vip') DEFAULT 'small' COMMENT '会议室类型',
    status ENUM('normal', 'maintenance', 'disabled') DEFAULT 'normal' COMMENT '状态',
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
    updated_at DATETIME DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    deleted_at DATETIME NULL COMMENT '删除时间',
    INDEX idx_floor (floor),
    INDEX idx_building (building),
    INDEX idx_capacity (capacity),
    INDEX idx_status (status),
    INDEX idx_deleted (deleted_at)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='会议室表';
```

### 2.2 会议预约记录表

```sql
CREATE TABLE meetings (
    id BIGINT PRIMARY KEY AUTO_INCREMENT COMMENT '会议ID',
    title VARCHAR(200) NOT NULL COMMENT '会议标题',
    room_id BIGINT NOT NULL COMMENT '会议室ID',
    organizer_id BIGINT NOT NULL COMMENT '组织者ID',
    attendees JSON COMMENT '参会人员ID列表',
    start_time DATETIME NOT NULL COMMENT '开始时间',
    end_time DATETIME NOT NULL COMMENT '结束时间',
    meeting_type VARCHAR(50) COMMENT '会议类型',
    description TEXT COMMENT '会议描述',
    recurrence_rule VARCHAR(100) COMMENT '重复规则',
    status ENUM('scheduled', 'cancelled', 'completed', 'in_progress') DEFAULT 'scheduled' COMMENT '状态',
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
    updated_at DATETIME DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    deleted_at DATETIME NULL COMMENT '删除时间',
    INDEX idx_room_time (room_id, start_time, end_time),
    INDEX idx_organizer (organizer_id),
    INDEX idx_status (status),
    INDEX idx_start_time (start_time),
    INDEX idx_deleted (deleted_at),
    FOREIGN KEY (room_id) REFERENCES meeting_rooms(id)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='会议预约表';
```

### 2.3 考勤记录表

```sql
CREATE TABLE attendance_records (
    id BIGINT PRIMARY KEY AUTO_INCREMENT COMMENT '考勤记录ID',
    user_id BIGINT NOT NULL COMMENT '用户ID',
    attendance_date DATE NOT NULL COMMENT '考勤日期',
    check_in_time DATETIME COMMENT '签到时间',
    check_out_time DATETIME COMMENT '签退时间',
    work_duration INT COMMENT '工作时长（分钟）',
    status ENUM('normal', 'late', 'early_leave', 'absent', 'leave') DEFAULT 'normal' COMMENT '考勤状态',
    late_minutes INT DEFAULT 0 COMMENT '迟到分钟数',
    early_leave_minutes INT DEFAULT 0 COMMENT '早退分钟数',
    location VARCHAR(200) COMMENT '签到位置',
    device VARCHAR(50) COMMENT '签到设备',
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
    updated_at DATETIME DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    UNIQUE KEY uk_user_date (user_id, attendance_date),
    INDEX idx_user_date (user_id, attendance_date),
    INDEX idx_status (status),
    INDEX idx_check_in (check_in_time)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='考勤记录表';
```

### 2.4 请假记录表

```sql
CREATE TABLE leave_requests (
    id BIGINT PRIMARY KEY AUTO_INCREMENT COMMENT '请假ID',
    user_id BIGINT NOT NULL COMMENT '用户ID',
    leave_type ENUM('annual', 'sick', 'personal', 'other') NOT NULL COMMENT '请假类型',
    start_date DATE NOT NULL COMMENT '开始日期',
    end_date DATE NOT NULL COMMENT '结束日期',
    total_days DECIMAL(3,1) NOT NULL COMMENT '请假天数',
    reason TEXT COMMENT '请假原因',
    status ENUM('pending', 'approved', 'rejected', 'cancelled') DEFAULT 'pending' COMMENT '状态',
    approver_id BIGINT COMMENT '审批人ID',
    approved_at DATETIME COMMENT '审批时间',
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
    updated_at DATETIME DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    INDEX idx_user (user_id),
    INDEX idx_dates (start_date, end_date),
    INDEX idx_status (status)
```

## 三、统计视图设计

### 3.1 会议使用率统计视图

```sql
CREATE VIEW v_meeting_usage_stats AS
SELECT
    r.id AS room_id,
    r.name AS room_name,
    r.floor,
    r.capacity,
    COUNT(m.id) AS meeting_count,
    SUM(TIMESTAMPDIFF(MINUTE, m.start_time, m.end_time)) AS total_minutes,
    COUNT(DISTINCT DATE(m.start_time)) AS active_days,
    ROUND(
        SUM(TIMESTAMPDIFF(MINUTE, m.start_time, m.end_time)) /
        (DATEDIFF(NOW(), DATE_SUB(NOW(), INTERVAL 30 DAY)) * 8 * 60) * 100,
        2
    ) AS usage_rate
FROM meeting_rooms r
LEFT JOIN meetings m ON r.id = m.room_id
    AND m.status IN ('scheduled', 'completed')
    AND m.start_time >= DATE_SUB(NOW(), INTERVAL 30 DAY)
WHERE r.status = 'normal' AND r.deleted_at IS NULL
GROUP BY r.id, r.name, r.floor, r.capacity;
```

### 3.2 考勤统计视图

```sql
CREATE VIEW v_attendance_stats AS
SELECT
    u.id AS user_id,
    u.department_id,
    d.name AS department_name,
    ar.attendance_date,
    COUNT(ar.id) AS total_records,
    SUM(CASE WHEN ar.status = 'normal' THEN 1 ELSE 0 END) AS normal_count,
    SUM(CASE WHEN ar.status = 'late' THEN 1 ELSE 0 END) AS late_count,
    SUM(CASE WHEN ar.status = 'early_leave' THEN 1 ELSE 0 END) AS early_leave_count,
    SUM(CASE WHEN ar.status = 'absent' THEN 1 ELSE 0 END) AS absent_count,
    SUM(ar.late_minutes) AS total_late_minutes,
    AVG(ar.work_duration) AS avg_work_duration
FROM users u
LEFT JOIN attendance_records ar ON u.id = ar.user_id
LEFT JOIN departments d ON u.department_id = d.id
WHERE u.deleted_at IS NULL
GROUP BY u.id, u.department_id, d.name, ar.attendance_date;
```

## 四、索引优化

### 4.1 统计查询常用索引

| 表名 | 索引名 | 字段 | 用途 |
|------|--------|------|------|
| meetings | idx_stats_date | start_time, end_time | 会议统计查询 |
| attendance_records | idx_attendance_date | attendance_date | 考勤统计查询 |
| leave_requests | idx_leave_dates | start_date, end_date | 请假统计查询 |

### 4.2 分区策略建议

对于大数据量表（attendance_records, meetings），建议按月份分区：

```sql
-- 考勤记录表分区示例
ALTER TABLE attendance_records
PARTITION BY RANGE (TO_DAYS(attendance_date)) (
    PARTITION p202601 VALUES LESS THAN (TO_DAYS('2026-02-01')),
    PARTITION p202602 VALUES LESS THAN (TO_DAYS('2026-03-01')),
    PARTITION p202603 VALUES LESS THAN (TO_DAYS('2026-04-01')),
    PARTITION p_future VALUES LESS THAN MAXVALUE
);
```
