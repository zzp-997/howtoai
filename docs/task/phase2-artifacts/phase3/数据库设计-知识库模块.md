## 一、设计概述

## 二、表结构设计

### 2.1 知识分类表

```sql
CREATE TABLE knowledge_categories (
    id BIGINT PRIMARY KEY AUTO_INCREMENT COMMENT '分类ID',
    name VARCHAR(100) NOT NULL COMMENT '分类名称',
    parent_id BIGINT NULL COMMENT '父级分类ID',
    level INT DEFAULT 1 COMMENT '分类层级（1-3）',
    sort_order INT DEFAULT 0 COMMENT '排序权重',
    icon VARCHAR(50) COMMENT '分类图标',
    description VARCHAR(500) COMMENT '分类描述',
    created_by BIGINT NOT NULL COMMENT '创建人ID',
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
    updated_at DATETIME DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    deleted_at DATETIME NULL COMMENT '删除时间',
    INDEX idx_parent (parent_id),
    INDEX idx_level (level),
    INDEX idx_sort (sort_order),
    INDEX idx_deleted (deleted_at),
    INDEX idx_created_by (created_by),
    FOREIGN KEY (parent_id) REFERENCES knowledge_categories(id)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='知识分类表';
```

### 2.2 知识内容表

```sql
CREATE TABLE knowledges (
    id BIGINT PRIMARY KEY AUTO_INCREMENT COMMENT '知识ID',
    title VARCHAR(200) NOT NULL COMMENT '文档标题',
    content LONGTEXT NOT NULL COMMENT '文档内容（富文本HTML）',
    summary VARCHAR(500) COMMENT '文档摘要',
    category_id BIGINT COMMENT '分类ID',
    author_id BIGINT NOT NULL COMMENT '作者ID',
    tags JSON COMMENT '标签JSON数组',
    visibility ENUM('public', 'department', 'custom') DEFAULT 'public' COMMENT '可见范围',
    custom_users JSON COMMENT '自定义可见用户ID列表',
    custom_departments JSON COMMENT '自定义可见部门ID列表',
    view_count INT DEFAULT 0 COMMENT '阅读次数',
    unique_view_count INT DEFAULT 0 COMMENT '唯一阅读人数',
    like_count INT DEFAULT 0 COMMENT '点赞次数',
    comment_count INT DEFAULT 0 COMMENT '评论次数',
    status ENUM('draft', 'published', 'archived') DEFAULT 'draft' COMMENT '状态',
    published_at DATETIME COMMENT '发布时间',
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
    updated_at DATETIME DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    deleted_at DATETIME NULL COMMENT '删除时间',
    INDEX idx_category (category_id),
    INDEX idx_author (author_id),
    INDEX idx_status (status),
    INDEX idx_visibility (visibility),
    INDEX idx_published (published_at),
    INDEX idx_view_count (view_count),
    INDEX idx_deleted (deleted_at),
    FULLTEXT INDEX ft_title_content (title, content),
    FOREIGN KEY (category_id) REFERENCES knowledge_categories(id),
    FOREIGN KEY (author_id) REFERENCES users(id)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='知识内容表';
```

### 2.3 知识阅读记录表

```sql
CREATE TABLE knowledge_views (
    id BIGINT PRIMARY KEY AUTO_INCREMENT COMMENT '阅读记录ID',
    knowledge_id BIGINT NOT NULL COMMENT '知识ID',
    user_id BIGINT NOT NULL COMMENT '用户ID',
    view_duration INT DEFAULT 0 COMMENT '阅读时长（秒）',
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
    UNIQUE KEY uk_knowledge_user (knowledge_id, user_id),
    INDEX idx_knowledge (knowledge_id),
    INDEX idx_user (user_id),
    INDEX idx_created (created_at),
    FOREIGN KEY (knowledge_id) REFERENCES knowledges(id),
    FOREIGN KEY (user_id) REFERENCES users(id)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='知识阅读记录表';
```

### 2.4 知识点赞表

```sql
CREATE TABLE knowledge_likes (
    id BIGINT PRIMARY KEY AUTO_INCREMENT COMMENT '点赞ID',
    knowledge_id BIGINT NOT NULL COMMENT '知识ID',
    user_id BIGINT NOT NULL COMMENT '用户ID',
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
    UNIQUE KEY uk_knowledge_user (knowledge_id, user_id),
    INDEX idx_knowledge (knowledge_id),
    INDEX idx_user (user_id),
    FOREIGN KEY (knowledge_id) REFERENCES knowledges(id),
    FOREIGN KEY (user_id) REFERENCES users(id)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='知识点赞表';
```

### 2.5 知识评论表

```sql
CREATE TABLE knowledge_comments (
    id BIGINT PRIMARY KEY AUTO_INCREMENT COMMENT '评论ID',
    knowledge_id BIGINT NOT NULL COMMENT '知识ID',
    parent_id BIGINT NULL COMMENT '父评论ID',
    user_id BIGINT NOT NULL COMMENT '用户ID',
    content TEXT NOT NULL COMMENT '评论内容',
    like_count INT DEFAULT 0 COMMENT '点赞数',
    status ENUM('visible', 'hidden', 'deleted') DEFAULT 'visible' COMMENT '状态',
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
    updated_at DATETIME DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    INDEX idx_knowledge (knowledge_id),
    INDEX idx_parent (parent_id),
    INDEX idx_user (user_id),
    INDEX idx_status (status),
    FOREIGN KEY (knowledge_id) REFERENCES knowledges(id),
    FOREIGN KEY (parent_id) REFERENCES knowledge_comments(id),
    FOREIGN KEY (user_id) REFERENCES users(id)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='知识评论表';
```

## 三、视图设计

### 3.1 知识列表视图

```sql
CREATE VIEW v_knowledge_list AS
SELECT
    k.id,
    k.title,
    k.summary,
    k.category_id,
    kc.name AS category_name,
    k.author_id,
    u.name AS author_name,
    u.avatar AS author_avatar,
    k.tags,
    k.view_count,
    k.like_count,
    k.comment_count,
    k.status,
    k.published_at,
    k.created_at,
    k.updated_at
FROM knowledges k
LEFT JOIN knowledge_categories kc ON k.category_id = kc.id
LEFT JOIN users u ON k.author_id = u.id
WHERE k.deleted_at IS NULL AND k.status = 'published';
```
